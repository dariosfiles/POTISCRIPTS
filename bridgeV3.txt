local Library = loadstring(game:HttpGet("https://raw.githubusercontent.com/ZanoLeafVN/KavoMobileUI/main/KavoMobile.lua"))()
local Window = Library.CreateLib("OreoHub - Bridge To Brainrot [v3]", "Ocean")

-- Services
local TweenService = game:GetService("TweenService")
local Players = game:GetService("Players")

-- Variables
local player = Players.LocalPlayer
local searchTerm = ""
local NORMAL_TEXT = "grab"

-- Helper Function: Handles the Smooth TP, Freeze, Interact, and Return
local function stealBrainrot(targetSmallText)
    local character = player.Character
    local hrp = character and character:FindFirstChild("HumanoidRootPart")
    
    if not hrp then return end

    -- 1. Save original position
    local originalCFrame = hrp.CFrame

    -- 2. Search for the prompt (Looks through all folders/descendants)
    local matches = {}
    for _, obj in ipairs(workspace:GetDescendants()) do
        if obj:IsA("ProximityPrompt") then
            local objText = string.lower(obj.ObjectText)
            local actText = string.lower(obj.ActionText)
            
            if objText == string.lower(targetSmallText) and actText == NORMAL_TEXT then
                table.insert(matches, obj)
            end
        end
    end

    -- 3. Execution logic
    if #matches > 0 then
        local randomPrompt = matches[math.random(1, #matches)]
        local targetPart = randomPrompt.Parent
        
        if targetPart:IsA("BasePart") then
            -- Smooth TP (0.3 seconds)
            local tweenInfo = TweenInfo.new(0.3, Enum.EasingStyle.Quad, Enum.EasingDirection.Out)
            local goal = {CFrame = targetPart.CFrame * CFrame.new(0, 3, 0)}
            local tween = TweenService:Create(hrp, tweenInfo, goal)
            
            tween:Play()
            tween.Completed:Wait()

            -- Freeze when you arrive
            hrp.Anchored = true
            
            -- Use the prompt
            task.wait(0.1)
            randomPrompt:InputHoldBegin()
            task.wait(randomPrompt.HoldDuration)
            randomPrompt:InputHoldEnd()
            
            -- Buffer wait so the server registers the grab
            task.wait(1)
            
            -- Return home and unfreeze
            hrp.Anchored = false
            hrp.CFrame = originalCFrame
            
            print("Successfully processed: " .. targetSmallText)
        end
    else
        warn("No prompts found for: " .. targetSmallText)
    end
end

-- Tab 1: Presets
local Tab = Window:NewTab("Stealing")
local SectionSteal = Tab:NewSection("Steals Brainrots")
SectionSteal:NewLabel("BETA VERSION")

SectionSteal:NewButton("Steal Garama", "Steals that brainrot", function()

SectionSteal:NewButton("Steal Los Matteos", "Steals that brainrot", function()

end)

end)

SectionSteal:NewButton("Steal God Block", "Steals that brainrot", function()
    stealBrainrot("brainrot god lucky block")
end)

-- Tab 2: Custom Search (Fixed Name so it shows up)
local TabBox = Window:NewTab("Custom Search")
local SectionStealBox = TabBox:NewSection("Manual Finder")

SectionStealBox:NewTextBox("BRAINROT NAME", "Type here the brainrot to search", function(txt)
    searchTerm = txt
end)

SectionStealBox:NewButton("GO FIND! âœ…", "Searches for that brainrot", function()
    if searchTerm ~= "" then
        stealBrainrot(searchTerm)
    else
        print("TextBox is empty!")
    end
end)
