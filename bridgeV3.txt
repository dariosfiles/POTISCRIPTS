local Library = loadstring(game:HttpGet("https://raw.githubusercontent.com/ZanoLeafVN/KavoMobileUI/main/KavoMobile.lua"))()
local Window = Library.CreateLib("Bridge For Brainrot | V4.5", "BloodTheme")

-- Services
local TweenService = game:GetService("TweenService")
local Players = game:GetService("Players")

-- Variables
local player = Players.LocalPlayer
local searchTerm = ""
local NORMAL_TEXT = "grab"

-- Helper Function (To avoid repeating the same code 5 times)
local function stealBrainrot(targetSmallText)
    local character = player.Character
    local hrp = character and character:FindFirstChild("HumanoidRootPart")
    if not hrp then return end

    local originalCFrame = hrp.CFrame
    local matches = {}

    for _, obj in ipairs(workspace:GetDescendants()) do
        if obj:IsA("ProximityPrompt") then
            if string.lower(obj.ObjectText) == string.lower(targetSmallText) and string.lower(obj.ActionText) == NORMAL_TEXT then
                table.insert(matches, obj)
            end
        end
    end

    if #matches > 0 then
        local randomPrompt = matches[math.random(1, #matches)]
        local targetPart = randomPrompt.Parent
        if not targetPart:IsA("BasePart") then return end

        local tween = TweenService:Create(hrp, TweenInfo.new(0.3, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {CFrame = targetPart.CFrame * CFrame.new(0, 3, 0)})
        tween:Play()
        tween.Completed:Wait()

        hrp.Anchored = true
        task.wait(0.1)
        randomPrompt:InputHoldBegin()
        task.wait(randomPrompt.HoldDuration)
        randomPrompt:InputHoldEnd()
        
        task.wait(1)
        hrp.Anchored = false
        hrp.CFrame = originalCFrame
    else
        print("Not found: " .. targetSmallText)
    end
end

-- TAB 1: PRESETS
local Tab = Window:NewTab("Stealing")
local SectionSteal = Tab:NewSection("Steals Brainrots")
SectionSteal:NewLabel("BETA VERSION")

SectionSteal:NewButton("Steal Garama", "Steals that brainrot", function() stealBrainrot("garama and madundung") end)
SectionSteal:NewButton("Steal Los Matteos", "Steals that brainrot", function() stealBrainrot("los matteos") end)
SectionSteal:NewButton("Steal Secret Block", "Steals that brainrot", function() stealBrainrot("secret lucky block") end)
SectionSteal:NewButton("Steal God Block", "Steals that brainrot", function() stealBrainrot("brainrot god lucky block") end)

-- TAB 2: CUSTOM SEARCH (FIXED NAME)
local TabBox = Window:NewTab("Custom Search") 
local SectionStealBox = TabBox:NewSection("Manual Finder")

SectionStealBox:NewTextBox("BRAINROT NAME", "Type here the brainrot to search", function(txt)
    searchTerm = txt
end)

SectionStealBox:NewButton("GO FIND! âœ…", "Searches for that brainrot", function()
    if searchTerm ~= "" then
        stealBrainrot(searchTerm)
    end
end)
